在vue3.0中的写法我们可以将vue中的方法引入
import {onMounted,watch} from 'vue'
var app = {
    setup(){
        onMounted(()=>{

        })
        watch(()=>{

        })
    }
};

function useMouse(){
    const x;
    const y;
    const update=e=>{
        x=e.pageX;
        y=e.pageY;
    }
    onMounted(()=>{
         window.addEventListener('mousemove',update)
    });

}
